//to start the server
$mongod


//to cancel the session ctrl-C.  the find, findOne can not be executed, so start "mongod".

//change to root
$cd /
//create the data directory needed for mongodb
$sudo mkdir -p /data/db

//give the permissions to directory (s. mongodb.lock error)
sudo chown `id -u` /data/db


//to import the books from json file
mongoimport --host 127.0.0.1 --db quiz --collection books --jsonArray books.json
//or for questions:
mongoimport --host 127.0.0.1 --db quiz --collection questions test1.json
